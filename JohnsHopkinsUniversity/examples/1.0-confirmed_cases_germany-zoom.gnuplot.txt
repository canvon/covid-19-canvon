# Based on: ../OurWorldInData.org/examples/2.2-bestaetigte_Faelle_neue_bestaetigte_Faelle_Deutschland-zoom.gnuplot.txt
set terminal qt size 1024, 768
set datafile separator comma
set xdata time
set xtics format "%F"
set y2tics
TC(Dummy) = timecolumn(1,"%s")
SelectLocation(Location, Value) = stringcolumn("location") eq Location ? Value : NaN
plot [strptime("%Y-%m-%d", "2020-02-27"):] 'confirmed_for_gnuplot.csv' using (TC(0)):(SelectLocation("Germany", column("total"))) title "bestaetigte Faelle Deutschland" with lines, '' using (TC(0)):(SelectLocation("Germany", column("new"))) axis x1y2 title "neue bestaetigte Faelle Deutschland" with lines
